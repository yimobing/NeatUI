SEO建站页面

包含：建站过程中SEO优化常用的meta、link标签，移动端适配等

                <head>
                    <meta charset="UTF-8">
                    <meta name="viewport" content="width=device-width,initial-scale=1.0,minimum-scale=1.0,maximum-scale=1.0,user-scalable=no">
                    <meta name="format-detection" content="telephone=no,email=no,date=no,address=no">
                    <meta name="apple-mobile-web-app-status-bar-style" content="black">
                    <meta name="apple-mobile-web-app-capable" content="yes">
                    <meta name="applicable-device" content="pc,mobile"><!--[告诉百度屏幕适配。值："pc" 仅pc版, "mobile" 仅移动版, "pc,mobile" 响应式]-->
                    <meta name="application-name" content="网站名称">
                    <meta name="author" content="网页作者">
                    <meta name="copyright" content="版权信息">
                    <!--[浏览器内核控制]-->
                    <meta name="renderer" content="webkit">
                    <meta http-equiv="X-UA-Compatible" content="IE=edge,chrome=1">
                    <!--[SEO优化1 + 移动适配]-->
                    <meta http-equiv="Cache-Control" content="no-transform"> <!--[禁止百度转码]-->
                    <meta http-equiv="Cache-Control" content="no-siteapp"> <!--[禁止百度转码]-->
                    <meta http-equiv="mobile-agent" content="format=html5;url=http://m.abc.com/"><!--移动适配：填写移动版对应URL-->
                    <!--[SEO优化2：社会化标签og：同意网页内容可被社会化网站引用等，目前这种协议被360搜索引擎、SNS网站(如Fackbook、renren)采用]-->
                    <meta property="og:type" content="article"/><!--页面类型，如：article、news、book、movie、product-->
                    <meta property="og:image" content="略缩图地址"/>
                    <meta property="og:release_date" content="页面发布日期"/>
                    <meta property="og:title" content="页面标题"/>
                    <meta property="og:description" content="页面描述"/>
                    <meta property="og:url" content="页面地址"/>
                    <!--[网页标题、关键词、描述]-->
                    <title>SEO建站页面</title>
                    <meta name="keywords" content="">
                    <meta name="description" content="">
                    <!--[SEO优化3 + 移动适配]-->
                    <link rel="canonical" href="规范当前页面链接地址">
                    <link rel="alternate" href="http://m.abc.com/" media="only screen and(max-width: 640px)"><!--[PC版与移动版页面URL相互指向]-->
                    <script type="text/javascript" src="http://www.abc.com/uaredirect.js"></script>
                    <script type="text/javascript">uaredirect("http://m.abc.com/");</script>
                    <!--[样式文件]-->
                    <link type="text/css" rel="stylesheet" href="index.css" media="all"><!--[适用于所有设备]-->
                    <link type="text/css" rel="stylesheet" href="index-320-640.css" media="only screen and (min-width: 320px) and (max-width: 640px)"><!--[指定设备宽]-->
                    <!--[网页图标]-->
                    <link type="images/x-icon" rel="shortcut icon"favicon href="/favicon.ico">
                    <link type="images/png" rel="icon" href="/favicon.png">
                    <!--[脚本文件]-->
                    <script type="text/javascript">if(top.location != self.location){ top.location = self.location; }</script><!--[防止被他人IFRAME嵌入引用]-->
                    <!--[if lt IE 9]>
                        <script type="text/javascript" src="https://cdn.bootcdn.net/ajax/libs/html5shiv/r29/html5.min.js"></script>
                        <script type="text/javascript" src="https://cdn.bootcdn.net/ajax/libs/respond.js/1.4.2/respond.min.js"></script>
                        <script type="text/javascript" src="https://cdn.bootcdn.net/ajax/libs/rem/1.3.4/js/rem.min.js"></script>
                        <script type="text/javascript" src="https://cdn.bootcdn.net/ajax/libs/json2/20160511/json2.min.js"></script>
                    <![endif]-->
                    <base target="_blank"><!--[链接默认打开方式]-->
                </head>
