//———————————————————————————————————————————————— // 方法2：使用离线版的算法 离线转换算法出现轻微偏差是正常的，因为它是基于公开的近似公式实现的，和百度官方的加密逻辑存在细微差异 //———————————————————————————————————————————————— /** * 离线版百度地图坐标转换： GCJ02 → BD09 * 几种坐标系：WGS84 : GPS 坐标，GCJ02 国测局火星坐标，微信、高德、高德地图), BD09 百度经纬度坐标 * @param {String} lng 经度 (GCJ02火星坐标) * @param {String} lat 纬度 (GCJ02火星坐标) */ function gcj02ToBd09Optimize(lng, lat) { // 调整前的公式 // var x_pi = 3.14159265358979324 * 3000.0 / 180.0; // var z = Math.sqrt(lng * lng + lat * lat) + 0.00002 * Math.sin(lat * x_pi); // var theta = Math.atan2(lat, lng) + 0.000003 * Math.cos(lng * x_pi); // var bdLng = z * Math.cos(theta) + 0.0065; // var bdLat = z * Math.sin(theta) + 0.006; // return { lng: bdLng, lat: bdLat }; // 调整后的公式 var x_pi = Math.PI * 3000.0 / 180.0; // 保留原逻辑基础上增加微小修正系数 var z = Math.sqrt(lng * lng + lat * lat) + 0.00002 * Math.sin(lat * x_pi); var theta = Math.atan2(lat, lng) + 0.00000301 * Math.cos(lng * x_pi); // 微调系数 var bd_lng = z * Math.cos(theta) + 0.006501; // 微调偏移量 var bd_lat = z * Math.sin(theta) + 0.00599; // 微调偏移量 return { longitude: bd_lng, latitude: bd_lat }; } 示例： // 腾讯地图坐标拾取器： https://lbs.qq.com/getPoint/ // 百度地图坐标拾取器： https://lbs.baidu.com/maptool/getpoint // 微信默认的坐标是调用腾讯地图 GCJ02 坐标 var gcjCoordinate = "118.608913, 24.909946"; // 原坐标。 GCJ02 坐标 var gcjArr = gcjCoordinate.split(','); var gcjLng = gcjArr[0], gcjLat = gcjArr[1]; // 经纬度 var bdCoordinate = gcj02ToBd09Optimize(gcjLng, gcjLat); // 百度地图坐标转换： GCJ02 → BD09 console.log('bdCoordinate：', bdCoordinate);
