一、快速上手

引入方式

方式一：单页面直接引入

<script src="./lib/axios.js"></script>
<script src="./lib/neatui-axios.js"></script>

方式二：模块化引入

<script type="module">
    import { getSomeData } from "../../api/public.js";
</script>

接口文件示例 public.js：

// 导入 neatui-axios，命名为 request
import request from "./neatui-axios.js";

// 配置项
var debug = true;
var method = 'post';
var baseURL = 'http://your-api.com';

/**
 * 获取某某数据
 * @param {Object} vm Vue实例对象
 * @param {Object} data 数据对象
 * @returns {Promise} 返回promise对象
 */
export function getSomeData(vm, data) {
    return request({
        debug: debug,
        vue: vm,
        original: false,
        method: method,
        baseURL: baseURL,
        url: "/api/jk_get_aaa.ashx",
        data: data
    });
}
说明：导入时命名为 request 而非 $axios，原因是 request 语义更清晰，表示"请求对象"，在接口文件中使用更自然，如 request.post()、request.get()。

调用方式

import { getSomeData } from "../../api/public.js";

getSomeData(this, { action: 'list' })
    .then(res => console.log('成功:', res))
    .catch(err => console.log('失败:', err.message));

配置参数（常用）

参数 类型 默认值 说明
debug Boolean false 是否启用调试模式
vue Object null Vue 实例对象，用于弹窗
original Boolean true 返回值是否为 axios 原生数据
popup Boolean true 出错时是否自动弹窗
baseURL String '' 接口基础地址
url String '' 接口地址
data Object {} POST 请求数据
timeout Number 5000 超时时间（毫秒）

二、详细说明

2.1 引入方式

方式一：单页面直接引入

适用于简单的 HTML 页面，无需构建工具。

<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>示例</title>
</head>
<body>
    <!-- 先引入 axios.js -->
    <script src="./lib/axios.js"></script>
    <!-- 再引入 neatui-axios.js -->
    <script src="./lib/neatui-axios.js"></script>
    
    <script>
        // 直接使用 $axios
        $axios.get('/api/user', {
            baseURL: 'http://your-api.com'
        }).then(res => {
            console.log('成功:', res);
        }).catch(err => {
            console.log('失败:', err.message);
        });
    </script>
</body>
</html>

方式二：模块化引入

适用于 Vue、React 等现代前端项目。

<script type="module">
    import { getSomeData } from "../../api/public.js";
</script>

接口文件示例 public.js：

// 导入 neatui-axios，命名为 request
import request from "./neatui-axios.js";

// 配置项
var debug = true;
var method = 'post';
var baseURL = 'http://your-api.com';

/**
 * 获取某某数据
 * @param {Object} vm Vue实例对象
 * @param {Object} data 数据对象
 * @returns {Promise} 返回promise对象
 */
export function getSomeData(vm, data) {
    return request({
        debug: debug,
        vue: vm,
        original: false,
        method: method,
        baseURL: baseURL,
        url: "/api/jk_get_aaa.ashx",
        data: data
    });
}
说明：导入时命名为 request 而非 $axios，原因是 request 语义更清晰，表示"请求对象"，在接口文件中使用更自然，如 request.post()、request.get()。

页面中调用：

import { getSomeData } from "../../api/public.js";

getSomeData(this, { action: 'list' })
    .then(res => console.log('成功:', res))
    .catch(err => console.log('失败:', err.message));

2.2 配置参数

自定义参数

参数 类型 默认值 说明
debug Boolean false 是否启用调试模式。true 时显示详细错误信息，false 时显示友好提示
vue Object null Vue 实例对象，用于调用 Element UI 或 Vant UI 的弹窗组件
original Boolean true 返回值是否为 axios 原生数据。true 返回完整响应，false 只返回 response.data
popup Boolean true 出错时是否自动弹出错误提示。设为 false 时需自行处理错误
loginTimeoutRedirectPage String '' 登录超时后跳转的页面地址

Axios 原生参数

参数 类型 默认值 说明
headers Object { "Content-Type": "application/json" } 请求头
method String 'get' 请求方法：get、post、put、delete 等
baseURL String '' 接口基础地址，会自动拼接在 url 前面
url String '' 接口地址
params Object {} GET 请求参数，会拼接到 URL 上
data Object {} POST/PUT 请求的数据体
timeout Number 5000 请求超时时间（毫秒），0 表示不限制

其它参数见 Axios 官网

2.3 调用方式

// 方式一：通用方式
request(options).then().catch();

// 方式二：POST
request.post(url, data, config).then().catch();

// 方式三：GET
request.get(url, config).then().catch();

// 方式四：并发请求
request.all([promise1, promise2]).then().catch();

POST 请求示例

request.post('/api/user', { name: 'test', age: 18 }, {
    baseURL: 'http://your-api.com',
    timeout: 10000
}).then(res => {
    console.log('成功:', res);
}).catch(err => {
    console.log('失败:', err.message);
});

GET 请求示例

request.get('/api/user', {
    baseURL: 'http://your-api.com',
    params: { id: 1 }
}).then(res => {
    console.log('成功:', res);
}).catch(err => {
    console.log('失败:', err.message);
});

并发请求示例

var request1 = request.get('/api/user');
var request2 = request.get('/api/order');

request.all([request1, request2]).then(res => {
    console.log('用户数据:', res[0]);
    console.log('订单数据:', res[1]);
}).catch(err => {
    console.log('失败:', err.message);
});

2.4 错误处理

错误返回值结构

{
    message: '错误提示信息',
    error: Error,           // 原始错误对象
    data: {},               // 请求数据
    isNetworkError: false,  // 是否网络错误
    isTimeout: false        // 是否超时
}

默认错误处理

默认情况下，请求失败会自动弹窗提示：

  • Element UI：调用 this.$alert() 弹窗
  • Vant UI：调用 this.$dialog.alert() 弹窗
  • 无 UI 框架：使用 alert() 弹窗

禁用自动弹窗

request.post(url, data, {
    popup: false
}).catch(err => {
    // 自行处理错误
    console.log(err.message);
});

常见错误类型

错误类型 说明
网络断开 isNetworkError: true，无法连接到服务器
请求超时 isTimeout: true，请求超过 timeout 时间
服务器错误 HTTP 状态码 5xx
客户端错误 HTTP 状态码 4xx（如 401 未授权、404 未找到）
业务错误 接口返回 return: 'fail'

2.5 Vue 项目示例

Vue 2 + Element UI

<template>
    <div>
        <el-button @click="fetchData">获取数据</el-button>
    </div>
</template>

<script>
import { getSomeData } from '@/api/public.js';

export default {
    methods: {
        async fetchData() {
            try {
                const res = await getSomeData(this, { action: 'list' });
                console.log('数据:', res);
            } catch (err) {
                console.log('错误:', err.message);
            }
        }
    }
};
</script>

Vue 2 + Vant UI

<template>
    <div>
        <van-button @click="fetchData">获取数据</van-button>
    </div>
</template>

<script>
import { getSomeData } from '@/api/public.js';

export default {
    methods: {
        async fetchData() {
            try {
                const res = await getSomeData(this, { action: 'list' });
                console.log('数据:', res);
            } catch (err) {
                console.log('错误:', err.message);
            }
        }
    }
};
</script>

Vue 3 + Element Plus

<template>
    <el-button @click="fetchData">获取数据</el-button>
</template>

<script setup>
import { ElMessage } from 'element-plus';
import { getSomeData } from '@/api/public.js';

const fetchData = async () => {
    try {
        // Vue 3 无 this，传 null，配合 popup: false 自行处理错误
        const res = await getSomeData(null, { action: 'list', popup: false });
        console.log('数据:', res);
    } catch (err) {
        ElMessage.error(err.message);
    }
};
</script>

Vue 3 + Vant

<template>
    <van-button @click="fetchData">获取数据</van-button>
</template>

<script setup>
import { showDialog } from 'vant';
import { getSomeData } from '@/api/public.js';

const fetchData = async () => {
    try {
        const res = await getSomeData(null, { action: 'list', popup: false });
        console.log('数据:', res);
    } catch (err) {
        showDialog({ message: err.message });
    }
};
</script>

Vue 2 与 Vue 3 使用差异

项目 Vue 2 Vue 3
选项式 API this 可直接传入 vue 参数 同 Vue 2
组合式 API - 无 this，vue 参数传 null，配合 popup: false 自行处理错误
Element UI 使用 this.$alert() 需手动引入 ElMessage、ElMessageBox
Vant 使用 this.$dialog.alert() 需手动引入 showDialog、showToast

2.6 常见问题

请求成功但进入 catch？

原因：接口返回格式不是标准格式 { return: "ok", data: ... }

解决：确认后端返回数据格式，或修改拦截器中的判断逻辑。

如何取消自动弹窗？

request.post(url, data, { popup: false }).then().catch();

如何获取完整的 axios 响应？

request.post(url, data, { original: true }).then(res => {
    console.log(res.status);   // 200
    console.log(res.data);     // 后端返回的数据
    console.log(res.headers);  // 响应头
});

如何处理登录超时自动跳转？

request.post(url, data, {
    vue: this,
    loginTimeoutRedirectPage: '/login.html'
});

当后端返回 return: 'fail' 且 data 包含 "登录超时" 或 "登陆超时" 时，会自动跳转到指定页面。

网络断开时页面假死？

请确保使用 v1.0.1 及以上版本，已修复此问题。
